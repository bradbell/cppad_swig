Octave: Size of Vectors: Example and Test
function ok = vector_size_xam()
     %
     % load the Cppad Swig library
     m_cppad
     %
     % initialize return variable
     ok = true;
     % -----------------------------------------------------------------------
     % create vectors
     bv = m_cppad.vec_bool();
     iv = m_cppad.vec_int(1);
     dv = m_cppad.vec_double(2);
     av = m_cppad.vec_a_double(3);
     %
     % check size of vectors
     ok = ok && bv.size() == 0 ;
     ok = ok && iv.size() == 1 ;
     ok = ok && dv.size() == 2 ;
     ok = ok && av.size() == 3 ;
     %
     return;
end
